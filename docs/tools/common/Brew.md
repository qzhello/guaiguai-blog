```shell
brew search  **  //查找某个软件包
brew list   //列出已经安装的软件的包
brew install  **  //安装某个软件包,默认安装的是稳定版本
brew uninstall  **  //卸载某个软件的包
brew upgrade  **  //更新某个软件包
brew info  **  //查看指定软件包的说明
brew cache clean  //清理缓存
```
